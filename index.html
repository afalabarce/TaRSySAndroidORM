TaRSySAndroidORM

Yet Another Advanced ORM for Android

View on GitHub Download .zip Download .tar.gz

Why I created this ORM if there are already many others?

I created this ORM for one simple reason ... I could not find one that would cover all my needs ...

I was looking for a ORM that was easy to deploy, powerful, and without any interaction by the developer at the time of making changes to the persistent entities.

How it works?

First... In your project add a reference to my orm project...

Next, in your Manifest file... Pay attention to meta-data tags into the application section...

    <application
        android:allowBackup="true"
        android:icon="@mipmap/ic_launcher"
        android:label="@string/app_name"
        android:supportsRtl="true"
        android:theme="@style/AppTheme" >

        <meta-data android:name="IS_EXTERNALSTORAGE" android:value="true" />

        <meta-data android:name="DATABASE_DIRECTORY" android:value="ExampleTaRSySORM" />

        <meta-data android:name="DATABASE_NAME" android:value="exampletarsysorm.db" />

        <meta-data android:name="ENTITY_PACKAGES" android:value="com.tarsys.exampletarsysorm.entities" />

        <meta-data android:name="DB_STRING_DEFAULT_LENGTH" android:value="500" />

        <activity android:name=".MainActivity" >

            <intent-filter>
                <action android:name="android.intent.action.MAIN" />

                <category android:name="android.intent.category.LAUNCHER" />

            </intent-filter>

        </activity>

    </application>

Next in your Main Activity (usually a Splash Activity) onCreate Method, you need to write this code...

    ...
    if (SGBDEngine.Initialize(this)){
    ... 
    }else{
       // if the initialization was not made...
    }
    ...

and... next, how we create the persistent entities? it's very simple... Let's look at a simple example...

    package com.tarsys.exampletarsysorm.entities;

    import com.tarsys.android.orm.annotations.DBEntity;
    import com.tarsys.android.orm.annotations.TableField;
    import com.tarsys.android.orm.enums.DBDataType;

    /**
     * Created by tarsys on 10/10/15.
     */
    @DBEntity(TableName = "person", Description = "person entity")
    public class Person {
        //region private vars
        private String personalIdentification;
        private String name;
        private String lastname1;
        private String lastname2;

        //endregion

        public Person (String identifierPK){
            this.personalIdentification = identifierPK;
        }    

        @TableField(FieldName = "personal_identification",
                Description = "Fiscal Number identification",
                DataType = DBDataType.StringDataType,
                DataTypeLength = 30,
                PrimaryKey = true,
                NotNull = true)
        public String getPersonalIdentification() {
            return personalIdentification;
        }
        public void setPersonalIdentification(String personalIdentification) {
            this.personalIdentification = personalIdentification;
        }

        @TableField(FieldName = "name",
            Description = "Person name",
            DataType = DBDataType.StringDataType,
            DataTypeLength = 120,
            NotNull = true,
            DefaultValue = "''")
        public String getName() {
            return name;
        }
        public void setName(String name) {
            this.name = name;
        }

        @TableField(FieldName = "lastname1",
            Description = "Last Name 1",
            DataType = DBDataType.StringDataType,
            DataTypeLength = 120,
            NotNull = true,
            DefaultValue = "''")
        public String getLastname1() {
            return lastname1;
        }
        public void setLastname1(String lastname1) {
            this.lastname1 = lastname1;
        }

        @TableField(FieldName = "lastname2",
            Description = "Last Name 2",
            DataType = DBDataType.StringDataType,
            DataTypeLength = 120,
            NotNull = true,
            DefaultValue = "''")
        public String getLastname2() {
             return lastname2;
        }

        public void setLastname2(String lastname2) {
            this.lastname2 = lastname2;
        }
    }

This example can be more simple...

    package com.tarsys.exampletarsysorm.entities;

    import com.github.tarsys.android.orm.annotations.DBEntity;
    import com.github.tarsys.android.orm.annotations.TableField;

    /**
     * Created by tarsys on 10/10/15.
     */
    @DBEntity
    public class Person {
        //region private vars
        private String personalIdentification;
        private String name;
        private String lastname1;
        private String lastname2;

        //endregion

        public Person(){

        }
        public Person(String identifier){
            this.personalIdentification = identifier;
        }

        @TableField(DataTypeLength = 30, PrimaryKey = true, NotNull = true)
        public String getPersonalIdentification() {
            return personalIdentification;
        }
        public void setPersonalIdentification(String personalIdentification) {
            this.personalIdentification = personalIdentification;
        }

        @TableField
        public String getName() {
            return name;
        }
        public void setName(String name) {
            this.name = name;
        }

        @TableField
        public String getLastname1() {
            return lastname1;
        }
        public void setLastname1(String lastname1) {
            this.lastname1 = lastname1;
        }

        @TableField
        public String getLastname2() {
            return lastname2;
        }

        public void setLastname2(String lastname2) {
            this.lastname2 = lastname2;
        }
    }

And... finally... how to use it?

    Person person = new Person();
    person.setPersonalIdentification("12345678Z");
    person.setName("Name");
    person.setLastname1("Last Name 1");
    person.setLastname2("Last Name 2");

    SGBDEngine.saveEntity(person);
    // Gets a list of persons (all persons)
    ArrayList<Person> persons = SGBDEngine.filterQuery(Person.class, null,null,null, true);
    if (persons != null){
        /// ...
    }

    // Get a person from his identifier...
    Person person1 = SGBDEngine.findPrimaryKeyEntity(new Person("12345678Z"), true);

How to integrate into android studio... i's very easy... using gradle, edit your build.gradle module and simply add this line:

compile 'com.github.tarsys.android:orm:1.0.0.2'

Note: if doesn't run directly with jcenter, try to change your project build.gradle, put your all projects section like this:

allprojects {
    repositories {
        mavenLocal()
        jcenter()
        maven {
            url  "http://dl.bintray.com/tarsys/maven"
        }
    }
}
